    **1. Create `src/bounds.rs`** with key matrix bounds validation:

    ```rust
    use crate::error::TransportError;
    use monsgeek_protocol::DeviceDefinition;

    /// Validate a key index and layer against device bounds.
    ///
    /// Returns Ok(()) if key_index < max_keys AND layer < max_layers.
    /// Returns Err(BoundsViolation) otherwise.
    ///
    /// This MUST be called before any SET_KEYMATRIX write to prevent
    /// firmware out-of-bounds memory corruption.
    pub fn validate_key_index(
        key_index: u16,
        max_keys: u16,
        layer: u8,
        max_layers: u8,
    ) -> Result<(), TransportError> {
        if key_index >= max_keys || layer >= max_layers {
            return Err(TransportError::BoundsViolation {
                key_index,
                max_keys,
                layer,
                max_layers,
            });
        }
        Ok(())
    }

    /// Validate a key matrix write request against a DeviceDefinition.
    ///
    /// Extracts key_count and layer from the device definition and delegates
    /// to validate_key_index. Returns Err if the device definition is missing
    /// key_count or layer fields (defensive: reject writes to unknown layouts).
    pub fn validate_write_request(
        device: &DeviceDefinition,
        key_index: u16,
        layer: u8,
    ) -> Result<(), TransportError> {
        let max_keys = device.key_count.ok_or_else(|| TransportError::BoundsViolation {
            key_index,
            max_keys: 0,
            layer,
            max_layers: 0,
        })? as u16;
        let max_layers = device.layer.ok_or_else(|| TransportError::BoundsViolation {
            key_index,
            max_keys,
            layer,
            max_layers: 0,
        })?;
        validate_key_index(key_index, max_keys, layer, max_layers)
    }
    ```

    **2. Create `deploy/99-monsgeek.rules`** udev rules file:

    ```
    # MonsGeek yc3121 keyboard udev rules for Linux
    #
    # Install: sudo cp 99-monsgeek.rules /etc/udev/rules.d/
    # Reload: sudo udevadm control --reload-rules && sudo udevadm trigger
    #
    # Purpose:
    # 1. Grant non-root USB access for MonsGeek keyboards (VID 0x3141)
    # 2. Unbind usbhid from IF2 (vendor interface) to prevent firmware STALL
    #
    # The yc3121 firmware STALLs on HID report descriptor reads for IF1/IF2,
    # causing 25-second timeouts and potentially resetting the USB port.
    # Unbinding usbhid from IF2 prevents the STALL while keeping IF0
    # (standard keyboard) functional under hid-generic.

    # Grant non-root USB access for MonsGeek VID via libusb
    SUBSYSTEM=="usb", ATTRS{idVendor}=="3141", MODE="0660", GROUP="plugdev", TAG+="uaccess"

    # Unbind usbhid from IF2 (vendor HID interface) to prevent firmware STALL
    # This fires after usbhid binds to the interface, then immediately unbinds it.
    # IF0 (keyboard) remains bound to hid-generic and works normally.
    ACTION=="add", SUBSYSTEM=="usb", DRIVER=="usbhid", ATTRS{idVendor}=="3141", ATTRS{idProduct}=="4005", ATTR{bInterfaceNumber}=="02", RUN+="/bin/sh -c 'echo -n %k > /sys/bus/usb/drivers/usbhid/unbind'"
    ```

    Note: The udev rule uses M5W's specific PID (0x4005) for the IF2 unbind rule. When new keyboards are added to the device registry, their PIDs should be added as additional rules. The non-root access rule uses only VID (0x3141) to cover all MonsGeek keyboards.
